def ===(other)
  # A case statement here causes and infinite loop
  # rubocop:disable Style/CaseLikeIf
  if self == ON
    other.on? if other.respond_to? :on?
  elsif self == OFF
    other.off? if other.respond_to?('off?')
  else
    false
  end
  # rubocop:enable Style/CaseLikeIf
end
