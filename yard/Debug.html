def debug_openhab
  logger.debug { "Things - Count #{things.size}" }
  things.each do |thing|
    logger.debug { "Thing:(#{thing.label})  UID:(#{thing.uid}) Channels(#{thing.channels.map(&:uid).join(', ')})" }
  end

end
