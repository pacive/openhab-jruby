def unit(unit)
  unit = SimpleUnitFormat.instance.unitFor(unit) if unit.is_a? String
  Thread.current.thread_variable_set(:unit, unit)
  yield
ensure
  Thread.current.thread_variable_set(:unit, nil)
end
