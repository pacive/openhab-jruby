def[](uid)
  thing_uid = ThingUID.new(*uid.split(':'))
  # rubocop: disable Style/GlobalVars
  thing = $things.get(thing_uid)
  # rubocop: enable Style/GlobalVars
  return unless thing

  logger.trace("Retrieved Thing(#{thing}) from registry for uid: #{uid}")
  Thing.new(thing)
end
